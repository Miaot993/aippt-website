export const proPosts = (await getCollection('docs'))
  .filter(post => post.id.startsWith('pro/') && !post.id.endsWith('index.mdx'))
  .sort((a, b) => {
    const dateA = a.data.date ? new Date(a.data.date).valueOf() : 0;
    const dateB = b.data.date ? new Date(b.data.date).valueOf() : 0;
    return dateB - dateA;
  });

export const featuredPosts = proPosts.filter(post => post.data.featured === true);
