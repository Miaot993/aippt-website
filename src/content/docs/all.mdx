import { getCollection } from 'astro:content';
import { Icon } from '@astrojs/starlight/components';
import ResourceCard from '../../components/ResourceCard.astro';

export const allPosts = (await getCollection('docs'))
  .filter(post => {
    {/* 🚨 修改 2：彻底重写过滤逻辑（因为 'pro/' 文件夹已经不存在了）*/}
    {/* 现在我们用“排除法”，把不需要显示在列表里的页面剔除 */}
    const isNotLite = !post.id.startsWith('lite/');
    const isNotOverview = post.id !== 'overview.mdx'; // 排除指挥中心
    const isNotSelf = !post.id.includes('all/index.mdx'); // 排除当前这个索引页自己
    const isNotRootIndex = post.id !== 'index.mdx'; // 排除你最外层的落地页首页

    // 只要不是上面这四个，统统视为 Pro Max 资源展示出来！
    return isNotLite && isNotOverview && isNotSelf && isNotRootIndex;
  })
  .sort((a, b) => {
    return new Date(b.data.date || 0) - new Date(a.data.date || 0);
  });

{/* ✅ 核心修正：严格复刻 Pro 首页的双层结构
  1. 外层 .not-content: 负责“清洗”样式，不带任何布局属性。
  2. 内层 div: 负责布局，只用 w-full max-w-7xl mx-auto px-6 lg:px-8。
     (去掉了之前的 mt-8，让它自然排列，和 Pro 首页完全零温差)
*/}
<div class="not-content">
  
  <div class="w-full max-w-7xl mx-auto">

    {/* 头部信息区：为了美观，我们在内部给它加一点上边距 (pt-8) 
        这样既不破坏外层容器的对齐逻辑，又能保持视觉呼吸感 
    */}
    <div class="flex flex-col items-start gap-4 mb-32 pt-0">
      <a href="/overview" class="inline-flex items-center gap-2 px-4 py-2 text-sm font-medium text-zinc-600 dark:text-zinc-400 bg-white dark:bg-zinc-900 border border-zinc-200 dark:border-zinc-800 rounded-lg hover:bg-zinc-50 dark:hover:bg-zinc-800 hover:text-indigo-600 dark:hover:text-indigo-400 transition-colors no-underline shadow-sm">
        <Icon name="left-arrow" class="w-4 h-4" />
        返回 Pro 用户专属页
      </a>
    </div>

    {/* 👇👇👇 2️⃣ 物理隔离带 (Spacer) 👇👇👇 */}
    {/* h-24 = 96px 的高度，你可以随意改为 h-32 (128px) 或 h-16 (64px) */}
    <div class="h-4 w-full"></div>

    {/* 卡片列表 */}
    <div class="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-6 pb-24">
      {
        allPosts.map((post) => (
          <ResourceCard 
            title={post.data.title}
            description={post.data.description}
            href={`/${post.slug}`} 
            date={post.data.date ? new Date(post.data.date).toLocaleDateString() : ''}
            
            /* 👇👇👇 核心修改在这里 👇👇👇 */
            /* 逻辑：如果这篇文章有 tags 且不为空，就显示tag；否则显示默认的 "资源" */
            tag={post.data.tags && post.data.tags.length > 0 ? post.data.tags : ["资源"]}
          />
        ))
      }
    </div>

  </div>
  
</div>
