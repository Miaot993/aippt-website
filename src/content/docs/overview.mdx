export function getHeroImage(post) {

  let imgPath = post.data.hero?.image?.file || post.data.banner;

    if (mdxImportMatch) imgPath = mdxImportMatch[1];

  // C. å…œåº•ç­–ç•¥ï¼šå°è¯•ç”¨æ­£åˆ™æŠ“å–æ­£æ–‡ä¸­çš„æ ‡å‡† Markdown å›¾ç‰‡è¯­æ³• (![alt](url))
  if (!imgPath) {

  }

  // D. ç»ˆæ­¢æ¡ä»¶ä¸å¤–é“¾æ”¾è¡Œ
  if (!imgPath) return null;
  if (imgPath.startsWith('http') || imgPath.startsWith('data:')) return imgPath;

  // E. æ ¸å¿ƒè·¯å¾„è½¬æ¢ï¼šå°†ç›¸å¯¹è·¯å¾„æ¸…ç†å¹¶æ‹¼è£…æˆå¸¦å“ˆå¸Œå€¼çš„ Vite ç»å¯¹è·¯å¾„
  const cleanImgPath = imgPath.replace(/^(\.\/|\/)/, ''); 
  const folderPath = post.id.split('/').slice(0, -1).join('/'); 
  const fullPath = `/src/content/docs/${folderPath ? folderPath + '/' : ''}${cleanImgPath}`;
  
  // F. ä»å…¨å±€å›¾ç‰‡æ± ä¸­æå‡ºçœŸå®çš„èµ„æºè·¯å¾„
  const imageModule = allImages[fullPath];

}

    return isNotLite && isNotSelf && isNotIndex;
