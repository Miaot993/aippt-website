export function getHeroImage(post) {
  // 1. ä¼˜å…ˆå°è¯•è·å– frontmatter é‡Œçš„å›¾ç‰‡ (å¦‚æœæœ‰çš„è¯)

  // ä¸“é—¨åŒ¹é…ç±»ä¼¼ `import bannerImg from "./01-images/banner.png";` çš„ä»£ç 
  if (!imgPath) {
    const mdxImportMatch = post.body?.match(/import\s+.*?\s+from\s+['"]([^'"]+\.(?:png|jpe?g|webp|gif|svg))['"]/i);
    if (mdxImportMatch) imgPath = mdxImportMatch[1]; // å®Œç¾æŠ“åˆ° "./01-images/banner.png"
  }

  // 3. å…œåº•ç­–ç•¥ï¼šæå–æ ‡å‡† Markdown å›¾ç‰‡ ![alt](url)
  if (!imgPath) {
    const mdMatch = post.body?.match(/!\[.*?\]\((.*?)\)/);

  }

  // 4. å¦‚æœå…¨éƒ½æ²¡æœ‰ï¼Œè¿”å› nullï¼Œéšè—å³ä¾§å›¾ç‰‡
  if (!imgPath) return null;

  const cleanImgPath = imgPath.replace(/^(\.\/|\/)/, '');

  const fullPath = `/src/content/docs/${folderPath ? folderPath + '/' : ''}${cleanImgPath}`;
  const imageModule = allImages[fullPath];
  
  return imageModule?.default?.src || imageModule?.default || imgPath;
}




export const proPosts = (await getCollection('docs'))
  .filter(post => {
    const isNotLite = !post.id.startsWith('lite/');
    const isNotSelf = post.id !== 'overview.mdx';
    const isNotIndex = !post.slug.endsWith('index');
    return isNotLite && isNotSelf && isNotIndex;
  })
  .sort((a, b) => {
    const dateA = new Date(a.data.date || '2000-01-01');
    const dateB = new Date(b.data.date || '2000-01-01');
    return dateB.valueOf() - dateA.valueOf();
  });

export const featuredPosts = proPosts.filter(post => post.data.featured === true);
